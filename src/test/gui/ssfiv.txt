(select-screen
  (name "ssfiv-select")
  (animation
    (depth background bottom)
    (image 0 "menu/paintown.png")
    (velocity -.001 0)
    (frame
      (image 0)
      (time -1)))
  
  ; Grid type
  ; - simple-list
  ; - grid-list
  (grid-list 
    (grid-size 12 3)
    (layout "static")
    (cell-dimensions 35 35)
    (cell-spacing -3 -1)
    (cell-margins 10 10)
    (cursors 2)
    (cursor-location 0 0)
    (cursor-location 1 11)
    (wrap true))
  ; Cells
  (cell (name back) (shape round) (radius 20) (color 0 0 0) (color-fill true))
  (cell (name top) (shape round) (radius 20) (color 255 255 255) (color-fill false))
  (cell (name select0) (shape round) (radius 20) (color 255 0 0) (color-fill false))
  (cell (name select1) (shape round) (radius 20) (color 0 0 255) (color-fill false))
  (cell (name select-alternative) (shape round) (radius 20) (color 0 255 0) (color-fill false))
  ; Auto populate the grid ?
  (auto-populate true)
  ; Directory in which to populate from
  (auto-populate-directory "players")
  ; Window coordinates where list will be
  ; x y width height
  (list-window 50 240 550 200)
  ; List depth
  (list-depth foreground top)
  ; Profile window is where the character avatar or animation can be drawn make one per cursor
  (profile-window 0 -10 320 480)
  (profile-window 320 -10 320 480 facing-left)
  ; Depth in which to display the profiles
  (profile-depth foreground bottom)
  ; Message Block they are added in the sequence that they are read, this can be useful for multiple selections (ie player1, then player2.. etc)
  (messages
    ; text Message
    (text (message "Character Select") (location 160 45) (color 0 255 255) (depth background top) (font-dimensions 40 40) (justification center))
    (text (message "Character Select") (location 480 45) (color 0 255 255) (depth background top) (font-dimensions 40 40) (justification center))
    (text (message "%s") (location 95 215) (color-low 0 0 0) (color-high 50 50 50) (interpolate-distance 50) (depth foreground bottom) (font-dimensions 35 35) (justification center) (profile-association 0))
    (text (message "%s") (location 90 210) (color 255 255 255) (depth foreground bottom) (font-dimensions 35 35) (justification center) (profile-association 0))
    (text (message "%s") (location 530 205) (color-low 0 0 0) (color-high 50 50 50) (interpolate-distance 50) (depth foreground bottom) (font-dimensions 35 35) (justification center) (profile-association 1))
    (text (message "%s") (location 525 200) (color 255 255 255) (depth foreground bottom) (font-dimensions 35 35) (justification center) (profile-association 1))
    )
  
  ; Add player by relative path
  ;(add "players/akuma/akuma.txt")
  
  ; Font
  ; (font somefont.ttf 32 32)
  ; Font dimensions
  (font-dimensions 35 35)
)
